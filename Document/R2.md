### 2. 编写 `main.rs` (核心初始化逻辑)
这段代码会打开一个窗口，并初始化 GPU。如果成功，你会看到一个**能够随窗口缩放、并被你用代码精确控制清除颜色的窗口**。

```rust
use winit::{
    event::*,
    event_loop::{ControlFlow, EventLoop},
    window::WindowBuilder,
};

struct State {
    surface: wgpu::Surface,
    device: wgpu::Device,
    queue: wgpu::Queue,
    config: wgpu::SurfaceConfiguration,
    size: winit::dpi::PhysicalSize<u32>,
    window: winit::window::Window,
}

impl State {
    // 初始化 GPU 状态
    async fn new(window: winit::window::Window) -> Self {
        let size = window.inner_size();

        // 1. Instance: 它是 wgpu 的入口，用来创建 Surface 和 Adapter
        let instance = wgpu::Instance::new(wgpu::InstanceDescriptor {
            backends: wgpu::Backends::all(), // 自动选择 Vulkan/DX12/Metal
            ..Default::default()
        });

        // 2. Surface: 它是我们绘图的“画布”，连接着窗口
        let surface = unsafe { instance.create_surface(&window) }.unwrap();

        // 3. Adapter: 物理显卡的抽象，用来检查显卡是否支持我们要的功能
        let adapter = instance.request_adapter(
            &wgpu::RequestAdapterOptions {
                power_preference: wgpu::PowerPreference::HighPerformance, // 追求极致性能
                compatible_surface: Some(&surface),
                force_fallback_adapter: false,
            },
        ).await.unwrap();

        // 4. Device & Queue: 
        // Device 是逻辑设备，用来创建 Shader、Buffer 等
        // Queue 是指令队列，用来把命令发给显卡
        let (device, queue) = adapter.request_device(
            &wgpu::DeviceDescriptor {
                features: wgpu::Features::empty(), // 暂时不需要特殊功能
                limits: wgpu::Limits::default(),   // 暂时使用默认限制
                label: None,
            },
            None,
        ).await.unwrap();

        // 5. Config: 配置 Surface 如何渲染（格式、尺寸）
        let surface_caps = surface.get_capabilities(&adapter);
        let surface_format = surface_caps.formats.iter()
            .copied()
            .filter(|f| f.is_srgb()) // 确保色彩空间正确
            .next()
            .unwrap_or(surface_caps.formats[0]);

        let config = wgpu::SurfaceConfiguration {
            usage: wgpu::TextureUsages::RENDER_ATTACHMENT, // 我们要往屏幕上画图
            format: surface_format,
            width: size.width,
            height: size.height,
            present_mode: wgpu::PresentMode::Fifo, // 开启垂直同步（防止撕裂）
            alpha_mode: surface_caps.alpha_modes[0],
            view_formats: vec![],
        };
        surface.configure(&device, &config);

        Self { window, surface, device, queue, config, size }
    }

    pub fn window(&self) -> &winit::window::Window { &self.window }

    // 处理窗口缩放
    pub fn resize(&mut self, new_size: winit::dpi::PhysicalSize<u32>) {
        if new_size.width > 0 && new_size.height > 0 {
            self.size = new_size;
            self.config.width = new_size.width;
            self.config.height = new_size.height;
            self.surface.configure(&self.device, &self.config);
        }
    }

    // 真正的渲染循环入口
    fn render(&mut self) -> Result<(), wgpu::SurfaceError> {
        // 获取当前帧的“画布”
        let output = self.surface.get_current_texture()?;
        let view = output.texture.create_view(&wgpu::TextureViewDescriptor::default());

        // 命令编码器：它记录我们要发给 GPU 的所有指令
        let mut encoder = self.device.create_command_encoder(&wgpu::GenericDescriptor { label: Some("Render Encoder") });

        {
            // 开启一个“渲染渲染通道”（Render Pass）
            let _render_pass = encoder.begin_render_pass(&wgpu::RenderPassDescriptor {
                label: Some("Render Pass"),
                color_attachments: &[Some(wgpu::RenderPassColorAttachment {
                    view: &view,
                    resolve_target: None,
                    ops: wgpu::Operations {
                        // 这一步就是清除背景色，你可以自己改颜色！
                        load: wgpu::LoadOp::Clear(wgpu::Color { r: 0.1, g: 0.2, b: 0.3, a: 1.0 }),
                        store: wgpu::StoreOp::Store,
                    },
                })],
                depth_stencil_attachment: None,
                occlusion_query_set: None,
                timestamp_writes: None,
            });
        }

        // 把指令提交给队列
        self.queue.submit(std::iter::once(encoder.finish()));
        output.present(); // 展示在屏幕上

        Ok(())
    }
}

fn main() {
    env_logger::init(); // 初始化日志
    let event_loop = EventLoop::new().unwrap();
    let window = WindowBuilder::new().with_title("My Style Engine").build(&event_loop).unwrap();

    // 使用 pollster 运行异步初始化
    let mut state = pollster::block_on(State::new(window));

    event_loop.run(move |event, _, control_flow| {
        match event {
            Event::WindowEvent { ref event, window_id } if window_id == state.window().id() => match event {
                WindowEvent::CloseRequested => *control_flow = ControlFlow::Exit,
                WindowEvent::Resized(physical_size) => state.resize(*physical_size),
                WindowEvent::ScaleFactorChanged { new_inner_size, .. } => state.resize(**new_inner_size),
                _ => {}
            },
            Event::RedrawRequested(window_id) if window_id == state.window().id() => {
                match state.render() {
                    Ok(_) => {}
                    Err(wgpu::SurfaceError::Lost) => state.resize(state.size),
                    Err(wgpu::SurfaceError::OutOfMemory) => *control_flow = ControlFlow::Exit,
                    Err(e) => eprintln!("{:?}", e),
                }
            }
            Event::MainEventsCleared => state.window().request_redraw(),
            _ => {}
        }
    }).unwrap();
}
```
